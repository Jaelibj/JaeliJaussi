import seaborn as sns

df_day= (
    campaign
      .groupby('day_of_week')
      .agg(
          total=('y', 'size'),
          y_yes=('y', lambda x: (x== "yes").sum())
      )
      .reset_index()
)
sns.barplot(
    data=df_day,
    x='day_of_week',
    y='y_yes'
)
df_day= (
    campaign[campaign['job']== 'student']

      )

)
